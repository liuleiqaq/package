
DECLARE @START  DATETIME   
DECLARE @END    DATETIME   
SET @START = '2018-07-28'   
SET @END = '2018-12-18'   
   
SELECT CONVERT(VARCHAR(7), DATEADD(MONTH,NUM,@START),120) AS  DateTimes   
FROM (SELECT ROW_NUMBER() OVER (ORDER BY ID)-1 AS NUM FROM SYSOBJECTS) KK  
WHERE CONVERT(VARCHAR(7), DATEADD(MONTH,NUM,@START),120) <=CONVERT(VARCHAR(7), @END,120)
