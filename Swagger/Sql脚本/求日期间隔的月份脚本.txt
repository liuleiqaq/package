--获取时间段间隔的月份
	DECLARE @START  DATETIME   
	DECLARE @END    DATETIME   
	SET @START = '2018-8-01'   
	SET @END = '2018-12-18'   
   
	SELECT CONVERT( VARCHAR(7),DATEADD(MONTH,NUM,@START),120) AS DateTimes INTO #tempData1  
	FROM (SELECT ROW_NUMBER() OVER (ORDER BY ID)-1 AS NUM FROM SYSOBJECTS) KK  
	WHERE DATEADD(MONTH,NUM,@START) <=@END
